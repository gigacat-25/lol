\begin{algorithm}
\begin{algorithmic}[1]
\caption{Euclid's algorithm}
\Procedure{Euclid}{$a,b$} \Comment{gcd of $a$ and $b$}
\State $r \gets a \bmod b$ 
\While{$r \neq 0$} \Comment{loop stops when r is 0}
\State $a \gets b$
\State $b \gets r$
\State $r \gets a \bmod b$
\EndWhile
\State \textbf{return} $b$ \Comment{the gcd is $b$}
\EndProcedure
\end{algorithmic}
\end{algorithm}
